ASP.NET Core Setup

Starter Program.cs example with PostgreSQL, CORS, controllers, and Swagger using variable naming.

Program.cs Snippet

using Microsoft.EntityFrameworkCore;
using variableapi.Model;
using variableapi.Persistence;

var builder = WebApplication.CreateBuilder(args);

// Add services to the container.
// Learn more about configuring Swagger/OpenAPI at https://aka.ms/aspnetcore/swashbuckle
builder.Services.AddEndpointsApiExplorer();
builder.Services.AddSwaggerGen();

builder.Services.AddDbContext<VariableDbContext>(options =>
{
    options.UseNpgsql(builder.Configuration.GetConnectionString("variabledb"));
});
builder.Services.AddTransient<VariableModel>();

builder.Services.AddCors(options =>
{
    options.AddPolicy("AllowFrontend",
        policy =>
        {
            policy
                .WithOrigins("http://localhost")
                .AllowAnyHeader()
                .AllowAnyMethod();
        });
});

builder.Services.AddControllers();

var app = builder.Build();

app.MapControllers();

app.UseCors("AllowFrontend");

// Configure the HTTP request pipeline.
if (app.Environment.IsDevelopment())
{
    app.UseSwagger();
    app.UseSwaggerUI();
}

//app.UseHttpsRedirection();
app.UseDefaultFiles();
app.UseStaticFiles();

app.Run();

Make sure appsettings.json has a ConnectionStrings:variabledb entry to match the snippet.

VariableModel Ops

CRUD

Model Base

using System.Data;
using Microsoft.EntityFrameworkCore.Storage;
using variableapi.Persistence;

namespace variableapi.Model;

public class VariableModel
{
    private readonly VariableDbContext _dbContext;

    public VariableModel(VariableDbContext dbContext)
    {
        _dbContext = dbContext;
    }
}

Get All Variables

public IEnumerable<GetAllVariablesDto> GetAllVariables()
{
    return _dbContext.variables.Select(x => new GetAllVariablesDto
    {
        VariableId = x.variableId,
        Title = x.title,
        Director = x.director,
        ReleaseYear = x.releaseYear,
        GroupId = x.groupId,
        GroupName = _dbContext.groups.Where(y => y.groupId == x.groupId).Select(y => y.name).First()
    });
}

Get Variable By Id

public IEnumerable<GetVariableByIdDto> GetVariableById(int id)
{
    var data = _dbContext.variables.Where(x => x.variableId == id);
    if (data == null)
    {
        throw new KeyNotFoundException();
    }
    else
    {
        return data.Select(x => new GetVariableByIdDto
            {
                Title = x.title,
                Director = x.director,
                ReleaseYear = x.releaseYear,
                GroupId = x.groupId,
                GroupName = _dbContext.groups.Where(y => y.groupId == x.groupId).Select(y => y.name).First()
            }
        );
    }
}

Create New Variable

public async Task CreateNewVariable(NewVariableDto data)
{
    if (data.ReleaseYear < 1800)
    {
        throw new ArgumentOutOfRangeException();
    }
    else if (!_dbContext.groups.Select(x => x.name).ToList().Contains(data.GroupName))
    {
        throw new KeyNotFoundException();
    }

    using var trx = _dbContext.Database.BeginTransaction();
    {
        _dbContext.Add(new VariableDbContext.Variable
        {
            title = data.Title,
            director = data.Director,
            releaseYear = data.ReleaseYear,
            groupId = _dbContext.groups.Where(x => x.name == data.GroupName).First().groupId,
        });
        _dbContext.SaveChanges();
        trx.Commit();
    }
}

Update Variable

public async Task UpdateVariable(UpdateVariableDto data)
{
    var updated = _dbContext.variables.Where(x => x.variableId == data.variableId).First();
    if (updated == null)
    {
        throw new KeyNotFoundException();
    }
    else if (data.groupId == null || data.Title == null || data.releaseYear == null || data.groupId == null)
    {
        throw new DataException();
    }
    else
    {
        using var trx = _dbContext.Database.BeginTransaction();
        {
            updated.title = data.Title;
            updated.director = data.Director;
            updated.releaseYear = data.releaseYear;
            updated.groupId = data.groupId;
        }
        _dbContext.SaveChanges();
        trx.Commit();
    }
}

Delete Variable

public async Task DeleteVariable(int id)
{
    if (id < 0)
    {
        throw new ArgumentOutOfRangeException();
    }
    else
    {
        var todelete = _dbContext.variables.Where(x => x.variableId == id).First();
        if (todelete == null)
        {
            throw new KeyNotFoundException();
        }
        else
        {
            using var trx = _dbContext.Database.BeginTransaction();
            {
                _dbContext.variables.Remove(todelete);
                _dbContext.SaveChanges();
                trx.Commit();
            }
        }
    }
}

Get All Groups

public IEnumerable<GetAllGroupsDto> GetAllGroups()
{
    return _dbContext.groups.Select(x => new GetAllGroupsDto
    {
        GroupId = x.groupId,
        Name = x.name,
        yearLimit = x.ageLimit,
        numberOfVariables = _dbContext.variables.Where(y => y.groupId == x.groupId).Count()
    });
}

Create Group

public async Task CreateGroup(CreateGroupDto data)
{
    if (data.name == null || data.ageLimit == null)
    {
        throw new DataException();
    }
    else
    {
        VariableDbContext.Group newGroup = new VariableDbContext.Group
        {
            ageLimit = data.ageLimit,
            name = data.name
        };
        using var trx = _dbContext.Database.BeginTransaction();
        {
            _dbContext.groups.Add(newGroup);
            _dbContext.SaveChanges();
            trx.Commit();
        }
    }
}

using System.Data;
using Microsoft.AspNetCore.Http;
using Microsoft.AspNetCore.Mvc;
using variableapi.Model;
using variableapi.Persistence;

namespace variableapi.Controller
{
    [Route("api/[controller]")]
    [ApiController]
    public class api : ControllerBase
    {
        private readonly VariableModel _variableModel;

        public api(VariableModel variableModel)
        {
            _variableModel = variableModel;
        }
    }
}

Get All Variables

[HttpGet("allvariables")]
public ActionResult<IEnumerable<GetAllVariablesDto>> GetAllVariables()
{
    try
    {
        return Ok(_variableModel.GetAllVariables());
    }
    catch (Exception e)
    {
        return BadRequest(e.Message);
    }
}

Get Variable By Id

Variable Entity

public class Variable
{
    [Key]
    [DatabaseGenerated(DatabaseGeneratedOption.Identity)]
    public int variableId { get; set; }
    public string title { get; set; }
    public string director { get; set; }
    public int releaseYear { get; set; }
    public int groupId { get; set; }
}
